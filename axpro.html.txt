<script>
  const S_ID = (typeof CONFIG !== 'undefined' && CONFIG.SHEET_ID) ? CONFIG.SHEET_ID : '';
  const A_URL = (typeof CONFIG !== 'undefined' && CONFIG.API_URL) ? CONFIG.API_URL : '';

  const userStr = sessionStorage.getItem('axUser');
  if(!userStr) {
    alert("로그인이 필요합니다.");
    location.replace('index.html');
  }
  const currentUser = JSON.parse(userStr);
  
  document.getElementById('loginUserInfo').textContent = `${currentUser.dept || ''} ${currentUser.name} ${currentUser.rank || ''}`;

  const isAxTeam = currentUser.dept && (currentUser.dept.includes('식품') || currentUser.dept.includes('환경'));
  const isManager = currentUser.rank && (currentUser.rank.includes('팀장') || currentUser.rank.includes('실장') || currentUser.rank.includes('사장') || currentUser.rank.includes('본부장'));
  
  // 팀원 여부 (관리자 제외 일반 AX팀 소속)
  const isRegularAxMember = isAxTeam && !isManager;

  let rawData = { members: [], projects: [], tasks: [] };
  
  let filters = { team: '전체', period: '전체', member: '전체', selectedProjects: new Set() };
  let isDockOpen = false;

  // 커스텀 드롭다운 선택값 저장소
  let cv = { memberFilter: '', inProject: '', inMember: '', edProject: '', edMember: '' };

  window.onload = async () => {
    if(!S_ID) return alert("config.js에 SHEET_ID가 없습니다.");
    await Promise.all([
      fetchData('members').then(d => rawData.members = d),
      fetchData('projects').then(d => rawData.projects = d),
      fetchData('tasks').then(d => rawData.tasks = d)
    ]);
    
    // RBAC 및 초기값 세팅 (로그인 시 내 이름으로 필터 고정)
    applyInitialSettings();

    // 드롭다운 리스트 렌더링
    renderProjectDropdowns();
    renderMemberDropdowns(); 
    
    updateProjectChips(); 
    renderBoard();
  };

  // 커스텀 드롭다운 제어 로직
  function toggleDropdown(id) {
    const wrapper = document.getElementById(`wrapper-${id}`);
    if (wrapper.classList.contains('disabled-dropdown')) return;

    const drop = document.getElementById(`drop-${id}`);
    const isHidden = drop.classList.contains('hidden');
    
    document.querySelectorAll('.dropdown-menu').forEach(el => el.classList.add('hidden'));
    
    if (isHidden) {
      drop.classList.remove('hidden');
      const input = drop.querySelector('input');
      input.value = ''; // 검색창 초기화
      filterList(id, ''); // 전체 항목 표시
      input.focus();
    }
  }

  function selectOption(id, value, text) {
    cv[id] = value;
    const display = document.getElementById(`display-${id}`);
    display.textContent = text || value || '선택';
    display.classList.remove('text-slate-400');
    display.classList.add('text-slate-800');

    document.getElementById(`drop-${id}`).classList.add('hidden');
    
    if (id === 'memberFilter') {
      filters.member = value === '' ? '전체' : value;
      updateProjectChips();
      renderBoard();
    }
  }

  function filterList(id, query) {
    const opts = document.getElementById(`opts-${id}`).children;
    const q = query.toLowerCase();
    for(let opt of opts) {
      const txt = opt.textContent.toLowerCase();
      opt.style.display = txt.includes(q) ? 'block' : 'none';
    }
  }

  // 바탕 클릭 시 드롭다운 닫기
  document.addEventListener('click', (e) => {
    if (!e.target.closest('.custom-select')) {
      document.querySelectorAll('.dropdown-menu').forEach(el => el.classList.add('hidden'));
    }
  });

  // 초기 권한 세팅 (RBAC 1~4번)
  function applyInitialSettings() {

  }

  function handleLogout() {

  async function fetchData(sheetName) {
    try {
      const res = await fetch(`https://docs.google.com/spreadsheets/d/${S_ID}/gviz/tq?tqx=out:csv&sheet=${sheetName}`);
      return parseCSV(await res.text());
    } catch(e) { console.error(e); return []; }
  }

  function parseCSV(t) {
    const lines=t.trim().split('\n'); if(lines.length<2) return [];
    const hdrs=lines[0].split(',').map(h=>h.trim().replace(/^"|"$/g,''));
    return lines.slice(1).map(l=>{
      const v=[]; let c='', q=false;
      for(let i=0; i<l.length; i++){ const ch=l[i]; if(ch==='"') q=!q; else if(ch===','&&!q){v.push(c.trim());c='';} else c+=ch; }
      v.push(c.trim()); const o={}; hdrs.forEach((h,i)=>o[h]=(v[i]||'').replace(/^"|"$/g,'')); return o;
    }).filter(row => Object.values(row).some(val => val !== '')); 
  }

  function renderProjectDropdowns() {
    const targetProjs = rawData.projects;
    const html = targetProjs.map(p => `<div class="px-3 py-2 hover:bg-blue-50 cursor-pointer text-[13px] text-slate-700 font-medium border-b border-slate-50 last:border-0" onclick="selectOption('__ID__', '${p.short_name}', '${p.short_name}')">${p.short_name}</div>`).join('');
    
    document.getElementById('opts-inProject').innerHTML = html.replace(/__ID__/g, 'inProject');
    document.getElementById('opts-edProject').innerHTML = html.replace(/__ID__/g, 'edProject');
  }

  function renderMemberDropdowns() {
    let allMembers = rawData.members.filter(m => m.dept && (m.dept.includes('식품') || m.dept.includes('환경')));
    
    // 필터 드롭다운용 (팀 필터에 따라 달라짐)
    let filterMembers = allMembers;
    if (filters.team !== '전체') {
      const teamKeyword = filters.team.replace('AX', '').trim();
      filterMembers = allMembers.filter(m => m.dept && m.dept.includes(teamKeyword));
    }
    
    const filterHTML = `<div class="px-3 py-2 hover:bg-blue-50 cursor-pointer text-[13px] font-black text-blue-600 border-b border-slate-100" onclick="selectOption('memberFilter', '', '전체')">전체</div>` 
                     + filterMembers.map(m => `<div class="px-3 py-2 hover:bg-blue-50 cursor-pointer text-[13px] text-slate-700 font-medium border-b border-slate-50 last:border-0" onclick="selectOption('memberFilter', '${m.name}', '${m.name}')">${m.name}</div>`).join('');
    document.getElementById('opts-memberFilter').innerHTML = filterHTML;

    // 추가/수정 드롭다운용 (팀에 무관하게 전사 직원 노출)
    const memHTML = allMembers.map(m => `<div class="px-3 py-2 hover:bg-blue-50 cursor-pointer text-[13px] text-slate-700 font-medium border-b border-slate-50 last:border-0" onclick="selectOption('__ID__', '${m.name}', '${m.name}')">${m.name}</div>`).join('');
    document.getElementById('opts-inMember').innerHTML = memHTML.replace(/__ID__/g, 'inMember');
    document.getElementById('opts-edMember').innerHTML = memHTML.replace(/__ID__/g, 'edMember');
  }

  function setFilter(type, value) {
    filters[type] = value;
    const parent = type === 'team' ? document.getElementById('teamFilters') : document.getElementById('periodFilters');
    parent.querySelectorAll('.chip').forEach(c => c.classList.remove('active'));
    event.target.classList.add('active');

    if(type === 'team') {
      renderMemberDropdowns();
      selectOption('memberFilter', '', '전체'); // 팀 변경 시 담당자 전체로 초기화
    }

    updateProjectChips();
    renderBoard();
  }

  function toggleAllProjects() {
    const targetProjs = filters.team === '전체' ? rawData.projects : rawData.projects.filter(p => p.team === filters.team);
    const selectableProjs = targetProjs.filter(p => {
      return rawData.tasks.some(t => 
        t.project_short === p.short_name &&
        (filters.period === '전체' || t.period === filters.period) &&
        (filters.member === '전체' || t.member_name === filters.member)
      );
    });
    
    let allSelected = true;
    if (selectableProjs.length === 0) allSelected = false;
    selectableProjs.forEach(p => {
      if (!filters.selectedProjects.has(p.short_name)) allSelected = false;
    });

    if (allSelected && selectableProjs.length > 0) filters.selectedProjects.clear();
    else selectableProjs.forEach(p => filters.selectedProjects.add(p.short_name));
    
    renderProjectChipsUI(); 
    renderBoard();
  }

  function toggleProjectChip(projName, chipElement) {
    if(chipElement.classList.contains('opacity-40')) return; 
    if (filters.selectedProjects.has(projName)) filters.selectedProjects.delete(projName);
    else filters.selectedProjects.add(projName);
    
    renderProjectChipsUI();
    renderBoard();
  }

  function renderProjectChipsUI() {
    const targetProjs = filters.team === '전체' ? rawData.projects : rawData.projects.filter(p => p.team === filters.team);
    const activeProjs = targetProjs.filter(p => {
      return rawData.tasks.some(t => 
        t.project_short === p.short_name &&
        (filters.period === '전체' || t.period === filters.period) &&
        (filters.member === '전체' || t.member_name === filters.member)
      );
    });
    
    const disabledProjs = targetProjs.filter(p => !activeProjs.includes(p));
    const sortedProjs = [...activeProjs, ...disabledProjs]; 
    
    const btnToggle = document.getElementById('btnToggleProjects');
    if (activeProjs.length === 0) {
      btnToggle.textContent = '선택 불가';
      btnToggle.classList.add('opacity-40', 'cursor-not-allowed');
      btnToggle.classList.remove('hover:bg-slate-300');
    } else if (activeProjs.every(ap => filters.selectedProjects.has(ap.short_name))) {
      btnToggle.textContent = '전체 해제';
      btnToggle.classList.remove('opacity-40', 'cursor-not-allowed');
      btnToggle.classList.add('hover:bg-slate-300');
    } else {
      btnToggle.textContent = '전체 선택';
      btnToggle.classList.remove('opacity-40', 'cursor-not-allowed');
      btnToggle.classList.add('hover:bg-slate-300');
    }

    document.getElementById('projectFilters').innerHTML = sortedProjs.map(p => {
      const isActive = activeProjs.includes(p);
      if (!isActive) {
        return `<button class="chip chip-multi opacity-40 cursor-not-allowed bg-slate-50 border-slate-200 text-slate-400" title="조건에 맞는 업무 없음">${p.short_name}</button>`;
      } else {
        const isSelected = filters.selectedProjects.has(p.short_name);
        return `<button class="chip chip-multi ${isSelected ? 'selected' : ''}" onclick="toggleProjectChip('${p.short_name}', this)">
                  ${isSelected ? '<span class="ms text-[14px]">check</span>' : ''} ${p.short_name}
                </button>`;
      }
    }).join('');
  }

  function updateProjectChips() {
    filters.selectedProjects.clear();
    const targetProjs = filters.team === '전체' ? rawData.projects : rawData.projects.filter(p => p.team === filters.team);
    targetProjs.forEach(p => {
      const hasTask = rawData.tasks.some(t => 
        t.project_short === p.short_name &&
        (filters.period === '전체' || t.period === filters.period) &&
        (filters.member === '전체' || t.member_name === filters.member)
      );
      if (hasTask) filters.selectedProjects.add(p.short_name);
    });
    renderProjectChipsUI();
  }

  function renderBoard() {
    const container = document.getElementById('boardContainer');
    const filteredTasks = rawData.tasks.filter(t => {
      if(filters.period !== '전체' && t.period !== filters.period) return false;
      if(filters.member !== '전체' && t.member_name !== filters.member) return false;
      if(filters.selectedProjects.size > 0 && !filters.selectedProjects.has(t.project_short)) return false;
      return true;
    });

    const grouped = {};
    filters.selectedProjects.forEach(proj => grouped[proj] = []); 
    filteredTasks.forEach(t => { if(grouped[t.project_short] !== undefined) grouped[t.project_short].push(t); });

    if(Object.keys(grouped).length === 0) {
      container.innerHTML = `<div class="text-center text-slate-400 font-bold mt-24 opacity-50 text-[14px]">해당 조건에 부합하는 과제 및 업무가 없습니다.</div>`;
      return;
    }

    container.innerHTML = Object.keys(grouped).map(projName => {
      const projInfo = rawData.projects.find(p => p.short_name === projName) || {};
      const tList = grouped[projName];
      const dailyTasks = tList.filter(t => t.period === '일간');
      const weeklyTasks = tList.filter(t => t.period === '주간');
      
      return `
      <section class="bg-white rounded-[20px] shadow-sm border border-slate-200 p-5 lg:p-7">
        <div class="flex items-center gap-3 mb-5 pb-4 border-b border-slate-100">
          <span class="bg-blue-50 text-[#0066cc] text-[11px] font-black px-2 py-0.5 rounded">${projInfo.team || '미분류'}</span>
          <h2 class="text-xl font-[800] text-slate-800 leading-none">${projName}</h2>
          <span class="text-[13px] text-slate-400 font-medium hidden sm:inline ml-1">- ${projInfo.full_name || ''}</span>
        </div>
        <div class="grid grid-cols-1 lg:grid-cols-2 gap-5 lg:gap-10">
          <div class="flex flex-col gap-3">
            <div class="text-[12px] font-black text-rose-500 uppercase flex items-center gap-1.5"><span class="ms text-[16px]">local_fire_department</span> Daily Focus</div>
            ${dailyTasks.length ? dailyTasks.map(t => generateTaskCard(t)).join('') : '<div class="text-[13px] text-slate-400 py-3 font-medium">등록된 일간 업무가 없습니다.</div>'}
          </div>
          <div class="flex flex-col gap-3 lg:border-l lg:border-slate-100 lg:pl-10">
            <div class="text-[12px] font-black text-[#0066cc] uppercase flex items-center gap-1.5"><span class="ms text-[16px]">calendar_today</span> Weekly Flow</div>
            ${weeklyTasks.length ? weeklyTasks.map(t => generateTaskCard(t)).join('') : '<div class="text-[13px] text-slate-400 py-3 font-medium">등록된 주간 업무가 없습니다.</div>'}
          </div>
        </div>
      </section>`;
    }).join('');
  }

  function generateTaskCard(t) {
    const pColor = t.priority === '상' ? 'text-rose-600 bg-rose-50' : (t.priority === '중' ? 'text-slate-500 bg-slate-100' : 'text-slate-400 bg-slate-50');
    let sColor = 'text-slate-600 bg-slate-50';
    if(t.status === '대기') sColor = 'text-emerald-600 bg-emerald-50';
    else if(t.status === '진행중') sColor = 'text-blue-600 bg-blue-50';
    else if(t.status === '문제발생') sColor = 'text-rose-600 bg-rose-50';
    else if(t.status === '완료') sColor = 'text-slate-400 bg-slate-100 line-through';
    
    return `
    <div class="task-card" onclick="openEditModal('${t.id}')">
      <div class="flex justify-between items-start gap-3 mb-2.5">
        <p class="text-[14px] font-semibold text-slate-700 leading-relaxed break-keep whitespace-pre-wrap">${t.content}</p>
        <span class="${pColor} text-[11px] font-bold px-2 py-0.5 rounded shrink-0">${t.priority}</span>
      </div>
      <div class="flex items-center gap-2.5">
        <span class="bg-slate-100 text-slate-600 text-[11px] font-bold px-2 py-1 rounded">${t.member_name}</span>
        ${t.progress ? `<div class="flex-1 max-w-[80px] h-[6px] bg-slate-100 rounded-full overflow-hidden"><div class="h-full bg-blue-500" style="width:${t.progress}%"></div></div><span class="text-[11px] font-bold text-slate-400">${t.progress}%</span>` : ''}
        <span class="ml-auto text-[11px] font-black ${sColor} px-2 py-1 rounded">${t.status}</span>
      </div>
    </div>`;
  }

  function toggleInputDock() {
    isDockOpen = !isDockOpen;
    document.getElementById('inputDockWrapper').classList.toggle('expanded');
    document.getElementById('dockIcon').style.transform = isDockOpen ? "rotate(45deg)" : "rotate(0deg)";
  }

  function addTask() {
    const proj = cv.inProject;
    const mem = cv.inMember; 
    const task = document.getElementById('inTask').value.trim();
    
    if(!proj || !task || !mem) return alert("과제명, 담당자, 업무 내용을 모두 선택/입력하세요.");
    
    const newTask = {
      id: 'T-' + Date.now(), project_short: proj, member_name: mem, content: task,
      period: document.getElementById('inPeriod').value, 
      status: document.getElementById('inStatus').value, 
      progress: document.getElementById('inProgress').value || 0, 
      priority: document.getElementById('inPriority').value,
      last_updated: new Date().toISOString()
    };
    
    rawData.tasks.push(newTask);
    document.getElementById('inTask').value = '';
    document.getElementById('inProgress').value = 0;
    document.getElementById('inStatus').value = '대기';
    document.getElementById('inPriority').value = '중';
    
    // 편의상 등록 후 담당자/과제명 초기화. (팀원은 내 이름 유지)
    if(isRegularAxMember) selectOption('inMember', currentUser.name, currentUser.name);
    else selectOption('inMember', '', '담당자 선택');
    selectOption('inProject', '', '과제명 선택');

    updateProjectChips(); renderBoard(); commitToCloud();
  }

  function openEditModal(id) {
    const t = rawData.tasks.find(x => x.id === id);
    if(!t) return;

    // RBAC: 팀원이 다른 사람의 업무를 수정하려 할 때 원천 차단 (경고창 디자인 개선)
    if(isRegularAxMember && t.member_name !== currentUser.name) {
      alert("🚫 [접근 제한] 타인 업무 수정 불가\n\n팀원은 본인이 작성한 업무만 수정 및 삭제가 가능합니다.\n\n✓ 팀원 : 본인 업무만 관리 가능\n✓ 관리자(팀장, 실장 등) : 모든 업무 관리 가능");
      return;
    }

    document.getElementById('edId').value = t.id;
    selectOption('edProject', t.project_short, t.project_short);
    selectOption('edMember', t.member_name, t.member_name);
    
    document.getElementById('edTask').value = t.content;
    document.getElementById('edStatus').value = t.status || '대기';
    document.getElementById('edProgress').value = t.progress || 0;
    document.getElementById('edPriority').value = t.priority || '중';
    document.getElementById('editModal').classList.add('active');
  }

  function closeEditModal() { document.getElementById('editModal').classList.remove('active'); }

  function saveEdit() {
    const id = document.getElementById('edId').value;
    const idx = rawData.tasks.findIndex(x => x.id === id);
    if(idx > -1) {
      rawData.tasks[idx] = {
        id: id, project_short: cv.edProject, member_name: cv.edMember, 
        content: document.getElementById('edTask').value,
        period: rawData.tasks[idx].period, status: document.getElementById('edStatus').value,
        progress: document.getElementById('edProgress').value, priority: document.getElementById('edPriority').value,
        last_updated: new Date().toISOString()
      };
      closeEditModal(); updateProjectChips(); renderBoard(); commitToCloud();
    }
  }

  function deleteTask() {
    if(!confirm("이 업무를 삭제하시겠습니까?")) return;
    const id = document.getElementById('edId').value;
    rawData.tasks = rawData.tasks.filter(x => x.id !== id);
    closeEditModal(); updateProjectChips(); renderBoard(); commitToCloud();
  }

  function commitToCloud() {
    if(!A_URL) return alert('API 주소가 설정되지 않았습니다.');
    if(window.isSaving) return;
    window.isSaving = true;

    document.getElementById('statusIndicator').querySelector('div').classList.replace('bg-emerald-500', 'bg-blue-500');
    document.getElementById('statusIndicator').querySelector('div').classList.add('syncing');
    document.getElementById('statusText').textContent = "Syncing...";

    const payload = { action: 'write', sheetName: 'tasks', data: rawData.tasks };
    document.getElementById('payload_input').value = JSON.stringify(payload);
    document.getElementById('hidden_form').action = A_URL;
    document.getElementById('hidden_form').submit();
  }

  function onSaveSuccess() {
    window.isSaving = false;
    document.getElementById('statusIndicator').querySelector('div').classList.replace('bg-blue-500', 'bg-emerald-500');
    document.getElementById('statusIndicator').querySelector('div').classList.remove('syncing');
    document.getElementById('statusText').textContent = "Connected";
  }

  function manualSync() { commitToCloud(); }
  window.addEventListener('keydown', (e) => { if ((e.ctrlKey || e.metaKey) && e.key === 's') { e.preventDefault(); manualSync(); } });
